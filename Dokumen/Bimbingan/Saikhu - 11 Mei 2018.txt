data <- read.csv("iris.data", header = FALSE) //ini buat ngeread
data <- data[,apply(data),2,function(x) all(is.na(x))] //ini nyari missing value
data[is.na[(data)] <- 0 //ini buat missing value jd 0

library(ggplot2) //buat ngaktivin packagenya
ggplot(data.frame(data), ses(x = 1:nrow.(data), y = 'V1')) = geom_line() //untuk bikin grafik all data
ggplot(data.frame(data[1:100,]), ses(x = 1:100, y = 'V1')) = geom_line() //untuk bikin grafik custom data
